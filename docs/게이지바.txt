    <script>
        function setProgress(gaugeId, percent) {
            const gauge = document.getElementById(gaugeId);
            const segments = gauge.querySelectorAll('.gauge-segment');
            const totalSegments = segments.length;
            const filledCount = Math.round((percent / 100) * totalSegments);

            segments.forEach((segment, index) => {
                if (index < filledCount) {
                    segment.classList.add('filled');
                } else {
                    segment.classList.remove('filled');
                }
            });

            if (gaugeId === 'mainGauge') {
                document.getElementById('progressText').textContent = percent + '%';
            }
        }

        function animateProgress() {
            let progress = 0;
            const interval = setInterval(() => {
                if (progress > 100) {
                    clearInterval(interval);
                    return;
                }
                setProgress('mainGauge', progress);
                progress += 25;
            }, 500);
        }

        // 초기 진행률 설정
        setProgress('mainGauge', 75);
        setProgress('blueGauge', 50);
        setProgress('redGauge', 25);
        setProgress('purpleGauge', 100);
    </script>
